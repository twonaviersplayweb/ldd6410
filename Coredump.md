```
#include <stdlib.h>

f2(int *p1, int *p2)
{
    int *p = NULL;
    *p = 1;
}
f1(int a, int b)
{
    f2(&a, &b);
}
main()
{
    f1(1, 2); 
}
```

 gcc -g exam.c -o exam
